### 1. AldonEnergyComp.cs (能量容器组件)

```csharp
using Verse;
using RimWorld;
using System.Collections.Generic;

namespace Aldon.Energy
{
    /// <summary>
    /// 能量容器组件 - 挂载在殖民者身上
    /// 管理: 容量、消耗、恢复、锁定
    /// </summary>
    public class AldonEnergyComp : ThingComp
    {
        // ===== 常量 =====
        public const int CAPACITY = 1000;
        public const int RECOVERY_RATE_PER_SECOND = 2;  // 每秒恢复2点

        // ===== 状态数据 =====
        private int usedEnergy = 0;      // 消耗待恢复的能量
        private int lockedEnergy = 0;    // 被装备占用的容量
        private Dictionary<string, int> locks = new Dictionary<string, int>();

        // ===== Tick驱动 =====
        private int tickCounter = 0;
        private const int TICKS_PER_RECOVERY = 30;  // 60 ticks/sec ÷ 2点/sec = 30 ticks/point

        // ===== 计算属性 =====
        public int Available => CAPACITY - lockedEnergy - usedEnergy;
        public int LockedEnergy => lockedEnergy;
        public int UsedEnergy => usedEnergy;
        public int CurrentCapacity => CAPACITY;

        public float RecoveryPercent => (float)(CAPACITY - usedEnergy) / CAPACITY;

        // ===== 生命周期 =====

        public override void CompTick()
        {
            base.CompTick();

            // 仅在能量未满时处理恢复
            if (usedEnergy > 0)
            {
                // 检查全局信号
                if (!AldonSignalManager.Instance.IsActive)
                {
                    return;  // 信号关闭时不恢复
                }

                // Tick累计，达到恢复间隔时执行恢复
                tickCounter++;
                if (tickCounter >= TICKS_PER_RECOVERY)
                {
                    AddEnergy(1);
                    tickCounter = 0;
                }
            }
        }

        // ===== 核心操作 =====

        /// <summary>
        /// 恢复能量
        /// </summary>
        public void AddEnergy(int amount)
        {
            if (amount <= 0) return;

            usedEnergy = System.Math.Max(0, usedEnergy - amount);

            if (parent is Pawn pawn)
            {
                Log.Message($"[Aldon] {pawn.Name} 恢复能量 +{amount}，当前可用: {Available}");
            }
        }

        /// <summary>
        /// 尝试消耗能量
        /// 返回: true=成功消耗, false=权限不足或能量不足
        /// </summary>
        public bool TryConsume(int amount)
        {
            if (amount <= 0)
            {
                Log.Warning("[Aldon] TryConsume: 消耗量必须 > 0");
                return false;
            }

            Pawn pawn = parent as Pawn;
            if (pawn == null) return false;

            // 检查权限
            AldonPermissionComp permComp = pawn.GetComp<AldonPermissionComp>();
            if (permComp == null || !permComp.HasPermission)
            {
                // 低权限消耗失败，但不显示错误（权限系统自己会提示）
                return false;
            }

            // 检查信号
            if (!AldonSignalManager.Instance.IsActive)
            {
                Messages.Message("驱动塔未激活，无法消耗能量", MessageTypeDefOf.RejectInput);
                return false;
            }

            // 检查可用能量
            if (Available < amount)
            {
                Messages.Message("能量不足", MessageTypeDefOf.RejectInput);
                return false;
            }

            // 执行消耗
            usedEnergy += amount;

            Log.Message($"[Aldon] {pawn.Name} 消耗能量 -{amount}，剩余: {Available}");
            return true;
        }

        /// <summary>
        /// 注册锁定 - 装备占用容量
        /// 返回: 锁定ID (用于后续释放)
        /// </summary>
        public string RegisterLock(int amount, string source)
        {
            if (amount <= 0)
            {
                Log.Warning("[Aldon] RegisterLock: 锁定量必须 > 0");
                return null;
            }

            if (lockedEnergy + amount > CAPACITY)
            {
                Pawn pawn = parent as Pawn;
                Log.Warning($"[Aldon] {pawn?.Name ?? "Unknown"} 锁定超出容量: 已占用{lockedEnergy} + 请求{amount} > {CAPACITY}");
                return null;
            }

            // 生成锁定ID
            string lockID = source + "_" + System.DateTime.Now.Ticks;

            // 记录
            locks[lockID] = amount;
            lockedEnergy += amount;

            Pawn p = parent as Pawn;
            Log.Message($"[Aldon] {p?.Name ?? "Unknown"} 注册锁定: {source} = {amount}点，ID: {lockID}");

            return lockID;
        }

        /// <summary>
        /// 释放锁定 - 装备卸下
        /// </summary>
        public bool ReleaseLock(string lockID)
        {
            if (locks.ContainsKey(lockID))
            {
                int amount = locks[lockID];
                lockedEnergy -= amount;
                locks.Remove(lockID);

                Pawn pawn = parent as Pawn;
                Log.Message($"[Aldon] {pawn?.Name ?? "Unknown"} 释放锁定: {lockID} = -{amount}点");
                return true;
            }

            return false;
        }

        // ===== 数据持久化 =====

        public override void PostExposeData()
        {
            base.PostExposeData();

            Scribe_Values.Look(ref usedEnergy, "usedEnergy", 0);
            Scribe_Values.Look(ref lockedEnergy, "lockedEnergy", 0);
            Scribe_Collections.Look(ref locks, "locks", LookMode.Value, LookMode.Value);

            if (locks == null)
            {
                locks = new Dictionary<string, int>();
            }
        }
    }
}
```

### 2. AldonPermissionComp.cs (权限组件)

```csharp
using Verse;
using RimWorld;

namespace Aldon.Energy
{
    public enum AldonPermissionLevel
    {
        Unauthorized = 0,   // Lv0 - 无权限
        Authorized = 1       // Lv1 - 授权用户
    }

    public class AldonPermissionComp : ThingComp
    {
        private AldonPermissionLevel permissionLevel = AldonPermissionLevel.Unauthorized;

        public AldonPermissionLevel PermissionLevel => permissionLevel;
        public bool HasPermission => permissionLevel >= AldonPermissionLevel.Authorized;

        // ===== 操作 =====

        public void GrantPermission()
        {
            permissionLevel = AldonPermissionLevel.Authorized;

            Pawn pawn = parent as Pawn;
            Messages.Message($"已赋予 {pawn.Name} Aldon能量系统权限", MessageTypeDefOf.PositiveEvent);
            Log.Message($"[Aldon] 赋予权限: {pawn.Name}");
        }

        public void RevokePermission()
        {
            permissionLevel = AldonPermissionLevel.Unauthorized;

            Pawn pawn = parent as Pawn;
            Messages.Message($"已撤销 {pawn.Name} 的 Aldon权限", MessageTypeDefOf.NeutralEvent);
            Log.Message($"[Aldon] 撤销权限: {pawn.Name}");
        }

        // ===== 数据持久化 =====

        public override void PostExposeData()
        {
            base.PostExposeData();
            Scribe_Values.Look(ref permissionLevel, "permissionLevel", AldonPermissionLevel.Unauthorized);
        }
    }
}
```

### 3. AldonSignalManager.cs (全局信号管理器)

```csharp
using Verse;
using System.Collections.Generic;

namespace Aldon.Energy
{
    /// <summary>
    /// 全局单例: 管理能量恢复信号
    /// 由驱动塔建筑控制激活/停用
    /// </summary>
    public class AldonSignalManager
    {
        private static AldonSignalManager instance;

        public static AldonSignalManager Instance
        {
            get
            {
                if (instance == null)
                {
                    instance = new AldonSignalManager();
                }
                return instance;
            }
        }

        private bool isActive = false;
        private List<System.Action> onActivated = new List<System.Action>();
        private List<System.Action> onDeactivated = new List<System.Action>();

        public bool IsActive => isActive;

        // ===== 控制 =====

        public void Activate()
        {
            if (isActive) return;

            isActive = true;
            Log.Message("[Aldon] 驱动塔已激活 - 能量恢复开始");

            // 广播回调
            foreach (var action in onActivated)
            {
                action?.Invoke();
            }
        }

        public void Deactivate()
        {
            if (!isActive) return;

            isActive = false;
            Log.Message("[Aldon] 驱动塔已停用 - 能量恢复停止");

            // 广播回调
            foreach (var action in onDeactivated)
            {
                action?.Invoke();
            }
        }

        public bool GetSignal() => isActive;

        // ===== 回调注册 =====

        public void RegisterOnActivated(System.Action callback)
        {
            if (!onActivated.Contains(callback))
            {
                onActivated.Add(callback);
            }
        }

        public void RegisterOnDeactivated(System.Action callback)
        {
            if (!onDeactivated.Contains(callback))
            {
                onDeactivated.Add(callback);
            }
        }

        public void UnregisterCallback(System.Action callback)
        {
            onActivated.Remove(callback);
            onDeactivated.Remove(callback);
        }
    }
}
```

### 4. Building_AldonDriver.cs (驱动塔建筑)

```csharp
using Verse;
using RimWorld;
using UnityEngine;
using System.Collections.Generic;

namespace Aldon.Energy
{
    public class Building_AldonDriver : Building
    {
        public override IEnumerable<Gizmo> GetGizmos()
        {
            foreach (Gizmo g in base.GetGizmos())
            {
                yield return g;
            }

            AldonSignalManager mgr = AldonSignalManager.Instance;

            if (!mgr.IsActive)
            {
                // 显示激活按钮
                yield return new Command_Action
                {
                    defaultLabel = "激活驱动塔",
                    defaultDesc = "启动能量恢复系统",
                    icon = TexCommand.AttackStabilize,  // 临时使用系统图标
                    action = () =>
                    {
                        AldonSignalManager.Instance.Activate();
                    }
                };
            }
            else
            {
                // 显示停用按钮
                yield return new Command_Action
                {
                    defaultLabel = "停用驱动塔",
                    defaultDesc = "关闭能量恢复系统",
                    icon = TexCommand.Halt,  // 临时使用系统图标
                    action = () =>
                    {
                        AldonSignalManager.Instance.Deactivate();
                    }
                };
            }
        }

        public override void Tick()
        {
            base.Tick();
            // 驱动塔本体的Tick逻辑（如电力检查等）在此添加
        }
    }
}
```

### 5. Patch_Verb_Cast.cs (Harmony Patch)

```csharp
using HarmonyLib;
using Verse;
using RimWorld;

namespace Aldon.Energy.Patches
{
    /// <summary>
    /// Patch: 在开枪完成后消耗能量
    /// 钩子点: Verb.Cast() 的 Postfix（开枪动作完成后）
    ///
    /// 设计说明：
    /// - 使用Postfix而不是Prefix，在开枪成功后再消耗能量
    /// - 如果能量不足，下次开枪时才会失败（不会阻止本次开枪）
    /// - 避免了TryStartCastOn的6参数签名问题
    /// - 权限检查在TryConsume中进行
    /// </summary>
    [HarmonyPatch(typeof(Verb), nameof(Verb.Cast))]
    public class Patch_Verb_CastPostfix
    {
        // 在Verb.Cast()执行完成后，尝试消耗能量
        static void Postfix(Verb __instance)
        {
            // 获取射手
            Pawn shooter = __instance.caster as Pawn;
            if (shooter == null) return;

            // 获取能量组件
            AldonEnergyComp energyComp = shooter.GetComp<AldonEnergyComp>();
            if (energyComp == null) return;

            // 尝试消耗能量（10点/发）
            // 如果消耗失败（权限不足或能量不足），显示消息但不阻止本次开枪
            if (!energyComp.TryConsume(10))
            {
                // TryConsume内部已经输出消息，这里不需要重复
                Log.Message("[Aldon] 能量消耗失败（权限或能量不足）");
            }
        }
    }
}
```

## 🐛 常见问题与解决方案

### Q1: "GetComp<T>() 返回null"
**原因**: Pawn没有这个组件
**解决**:
1. 检查Pawn的ThingDef是否在comps列表中配置了AldonEnergyComp
2. 使用 `if (comp == null) return;` 防守式检查

### Q2: "AddEnergy没有恢复能量"
**原因**: CompTick()没有被调用或信号关闭
**解决**:
1. 检查 AldonSignalManager.IsActive 是否为true
2. 检查 tickCounter 逻辑是否正确（应该是60/2=30 ticks才恢复1点）
3. 添加Log.Message()输出调试信息

### Q3: "TryConsume 总是返回false"
**原因**: 权限检查失败或能量不足
**解决**:
1. 先调用 GrantPermission() 赋予权限
2. 检查 Available >= 消耗量
3. 添加日志输出具体原因

### Q4: "Harmony Patch 没有生效"
**原因**: Patch的钩子点不正确或没有加载
**解决**:
1. 验证 [HarmonyPatch] 属性的类名和方法名是否正确
2. 在AldonMod的静态构造器中调用 `new Harmony("aldon.energy").PatchAll();`
3. 添加 Log.Message() 在Patch中确认执行

## ✅ 单元测试框架（可选但推荐）

在开发过程中，可以编写简单的测试来验证逻辑：

```csharp
// 测试文件: Tests/TestAldonEnergy.cs
public class TestAldonEnergy
{
    [Test]
    public void TestAddEnergy()
    {
        // 创建虚拟Pawn
        var pawn = new Pawn();
        var comp = new AldonEnergyComp { parent = pawn };

        // 测试恢复
        comp.usedEnergy = 100;
        comp.AddEnergy(10);
        Assert.AreEqual(90, comp.usedEnergy);
    }

    [Test]
    public void TestTryConsume()
    {
        var pawn = new Pawn();
        var energyComp = new AldonEnergyComp { parent = pawn };
        var permComp = new AldonPermissionComp { parent = pawn };

        // 模拟GetComp返回
        pawn.SetComp(energyComp);
        pawn.SetComp(permComp);

        // 无权限时应该失败
        Assert.IsFalse(energyComp.TryConsume(10));

        // 赋予权限
        permComp.GrantPermission();

        // 激活信号
        AldonSignalManager.Instance.Activate();

        // 现在应该成功
        Assert.IsTrue(energyComp.TryConsume(10));
        Assert.AreEqual(10, energyComp.UsedEnergy);
    }
}
```
