### 1.2 新建 TrionTrigger 类

**文件**：`ProjectTrion_MVP/TrionTrigger.cs`

```csharp
using System.Collections.Generic;
using Verse;
using RimWorld;

namespace ProjectTrion_MVP
{
    /// <summary>
    /// Trion 触发器装备 - 继承 Apparel（穿戴装备）
    ///
    /// 关键改动：
    /// - 改为 Apparel 而非 ThingWithComps
    /// - 穿戴关系由 Apparel 系统全自动管理
    /// - wearer 属性由 Apparel 提供
    /// </summary>
    public class TrionTrigger : Apparel
    {
        // ============ 数据 ============
        private List<TriggerMount> _mounts = new List<TriggerMount>();

        // ============ 属性 ============
        public List<TriggerMount> Mounts => _mounts;
        public int MountCount => _mounts.Count;

        // ============ 初始化 ============

        /// <summary>
        /// Apparel.Wear() 被调用时触发
        /// 用于初始化3个挂载点
        /// </summary>
        public override void Wear(Pawn pawn)
        {
            base.Wear(pawn);

            // 初始化挂载点（仅初始化一次）
            if (_mounts.Count == 0)
            {
                _mounts.Add(new TriggerMount(this, "LeftHand", 4));
                _mounts.Add(new TriggerMount(this, "RightHand", 4));
                _mounts.Add(new TriggerMount(this, "Special", 1));

                Log.Message("[Trion] 触发器已穿戴，3个挂载点初始化完成");
            }
        }

        // ============ 查询方法 ============

        /// <summary>
        /// 获取穿着者（Pawn）
        /// wearer 属性由 Apparel 系统提供
        /// </summary>
        public Pawn GetWearer()
        {
            return wearer;
        }

        /// <summary>
        /// 获取穿着者的 CompTrion
        /// </summary>
        public CompTrion GetWearerCompTrion()
        {
            return wearer?.GetComp<CompTrion>();
        }

        /// <summary>
        /// 根据槽位名称获取挂载点
        /// </summary>
        public TriggerMount GetMount(string slotName)
        {
            foreach (var mount in _mounts)
            {
                if (mount.SlotName == slotName)
                    return mount;
            }
            return null;
        }

        // ============ 序列化 ============

        /// <summary>
        /// 存档/读档处理
        ///
        /// Apparel 会自动序列化：
        /// - wearer 引用
        /// - 穿戴状态
        ///
        /// 我们只需要序列化自定义的 _mounts 列表
        /// </summary>
        public override void ExposeData()
        {
            base.ExposeData();
            Scribe_Collections.Look(ref _mounts, "mounts", LookMode.Deep);
        }
    }
}
```

**文件**：`ProjectTrion_MVP/TriggerComponent.cs`

```csharp
using Verse;

namespace ProjectTrion_MVP
{
    /// <summary>
    /// Trion 组件数据类 - 轻量级
    ///
    /// 关键改动：
    /// - 改为普通数据类，不继承任何游戏类
    /// - 引用 ThingDef 来获取定义数据
    /// - 参数通过 ModExtension 存储
    /// </summary>
    public class TriggerComponent : IExposable
    {
        // ============ 数据 ============
        private ThingDef _def;        // 引用 TriggerComponent_* 的定义
        private bool _isActive;       // 激活状态

        // ============ 属性 ============
        public ThingDef Def => _def;
        public bool IsActive => _isActive;

        /// <summary>
        /// 获取组件的 Trion 占用值
        /// </summary>
        public float ReservedCost
        {
            get
            {
                var ext = _def.GetModExtension<TriggerComponentExt>();
                return ext?.reservedCost ?? 10f;
            }
        }

        /// <summary>
        /// 获取组件每 Tick 的消耗
        /// </summary>
        public float TrionCost
        {
            get
            {
                var ext = _def.GetModExtension<TriggerComponentExt>();
                return ext?.trionCost ?? 1f;
            }
        }

        // ============ 构造函数 ============

        public TriggerComponent()
        {
            // 用于序列化
        }

        private TriggerComponent(ThingDef def)
        {
            _def = def;
            _isActive = false;
        }

        // ============ 创建方法 ============

        /// <summary>
        /// 从 defName 创建组件对象
        /// </summary>
        public static TriggerComponent CreateFromDef(string defName)
        {
            var def = DefDatabase<ThingDef>.GetNamed(defName);
            return new TriggerComponent(def);
        }

        /// <summary>
        /// 从 ThingDef 创建组件对象
        /// </summary>
        public static TriggerComponent CreateFromDef(ThingDef def)
        {
            return new TriggerComponent(def);
        }

        // ============ 状态方法 ============

        /// <summary>
        /// 设置激活状态
        /// </summary>
        public void SetActive(bool active)
        {
            _isActive = active;
            Log.Message($"[Trion] {_def.label} 设置激活状态: {active}");
        }

        // ============ 序列化 ============

        public void ExposeData()
        {
            // 只需要序列化 defName，读档时重建
            string defName = _def?.defName;
            Scribe_Values.Look(ref defName, "defName");

            if (Scribe.mode == LoadSaveMode.LoadingVars && defName != null)
            {
                _def = DefDatabase<ThingDef>.GetNamed(defName);
            }

            Scribe_Values.Look(ref _isActive, "isActive");
        }
    }

    /// <summary>
    /// TriggerComponent 的参数扩展
    ///
    /// 在 XML 的 ModExtensions 中定义参数：
    /// <li Class="ProjectTrion_MVP.TriggerComponentExt">
    ///     <reservedCost>10</reservedCost>
    ///     <trionCost>1</trionCost>
    /// </li>
    /// </summary>
    public class TriggerComponentExt : DefModExt
    {
        public float reservedCost = 10f;  // Trion 占用值
        public float trionCost = 1f;      // 每 Tick 消耗
    }
}
```

## 改动3：破裂处理与 Bail Out 成功传送

### 3.1 CompTrion 中的破裂处理方法

**新增/修改 CompTrion 中的方法**：

```csharp
// CompTrion.cs

/// <summary>
/// 触发破裂处理（被迫解除战斗体）
///
/// 触发条件：
/// - Available <= 0（能量耗尽）
/// - 玩家手动触发 Bail Out
/// - 供给器官被摧毁
///
/// 处理流程：
/// 1. 快照恢复
/// 2. Reserved → Consumed
/// 3. 传送（到锚点或基地）
/// 4. 施加 debuff
/// 5. 标记解除
/// </summary>
public void TriggerBailOut()
{
    if (!_isInCombat)
        return;

    Pawn pawn = parent as Pawn;

    // 1. 快照恢复
    if (_snapshot != null)
    {
        _snapshot.RestoreToPawn(pawn);
        Log.Message($"[Trion] {pawn.Name} 快照已恢复");
    }

    // 2. Reserved → Consumed（永久流失）
    _consumed += _reserved;
    _reserved = 0;

    // 3. 获取传送目标
    IntVec3 target = _strategy.GetBailOutTarget(pawn);

    if (target.IsValid)
    {
        // 执行传送
        pawn.Position = target;
        pawn.Notify_Teleported();
        Log.Message($"[Trion] {pawn.Name} 传送到 {target}");
    }

    // 4. 施加 debuff（Trion 枯竭）
    var hediffDef = DefDatabase<HediffDef>.GetNamed("Hediff_TrionExhausted", false);
    if (hediffDef != null)
    {
        var hediff = HediffMaker.MakeHediff(hediffDef, pawn);
        pawn.health.AddHediff(hediff);
        Log.Message($"[Trion] {pawn.Name} 施加了 Trion 枯竭 debuff");
    }

    // 5. 标记解除
    _isInCombat = false;
    Log.Message($"[Trion] {pawn.Name} 战斗体破裂，紧急脱离完成");
}

/// <summary>
/// 安全解除战斗体（主动）
///
/// 触发条件：
/// - Available > 0
/// - 玩家点击"解除战斗体"按钮
///
/// 处理流程：
/// 1. Reserved 完全返还
/// 2. 无快照恢复
/// 3. 无传送
/// 4. 无 debuff
/// </summary>
public void DestroyCombatBodySafely()
{
    if (!_isInCombat)
        return;

    Pawn pawn = parent as Pawn;

    if (Available <= 0)
    {
        // Available <= 0 时不能安全解除
        Log.Warning($"[Trion] {pawn.Name} Trion 能量不足，无法安全解除");
        return;
    }

    // 返还所有 Reserved
    _reserved = 0;

    // 标记解除（无其他处理）
    _isInCombat = false;
    Log.Message($"[Trion] {pawn.Name} 主动解除战斗体，无代价");
}
```

### 3.2 Strategy 中的传送逻辑

```csharp
// DefaultTrionStrategy.cs

public IntVec3 GetBailOutTarget(Pawn pawn)
{
    Map map = pawn.Map;

    if (map == null)
    {
        Log.Error("[Trion] 无法获取地图，传送失败");
        return IntVec3.Invalid;
    }

    // 优先级1：查询传送锚
    var anchors = map.listerThings.ThingsOfDef(
        DefDatabase<ThingDef>.GetNamed("Building_BailOutAnchor", false)
    );

    if (anchors != null && anchors.Count > 0)
    {
        Building_BailOutAnchor nearest = null;
        float nearestDist = float.MaxValue;

        foreach (var thing in anchors)
        {
            if (thing is Building_BailOutAnchor anchor && anchor.IsActive)
            {
                float dist = pawn.Position.DistanceTo(anchor.Position);
                if (dist < nearestDist)
                {
                    nearestDist = dist;
                    nearest = anchor;
                }
            }
        }

        if (nearest != null)
        {
            Log.Message($"[Trion] 传送到传送锚 @ {nearest.Position}");
            return nearest.Position;
        }
    }

    // 优先级2：传送到基地附近（参考空投逻辑）
    var safeSpot = RCellFinder.RandomSpotJustOutsideColony(map);

    Log.Message($"[Trion] 传送到基地附近 @ {safeSpot}");
    return safeSpot;
}
```

```csharp
// CompTrion.cs

public override void ExposeData()
{
    base.ExposeData();

    // 核心数据
    Scribe_Values.Look(ref _capacity, "capacity");
    Scribe_Values.Look(ref _reserved, "reserved");
    Scribe_Values.Look(ref _consumed, "consumed");
    Scribe_Values.Look(ref _isInCombat, "isInCombat");

    // 快照（完整序列化）
    Scribe_Deep.Look(ref _snapshot, "snapshot");

    // 组件列表（只存 defName）
    List<string> componentDefNames = new List<string>();

    if (Scribe.mode == LoadSaveMode.Saving)
    {
        // 保存：从穿戴的触发器提取所有组件 defName
        Pawn pawn = parent as Pawn;
        TrionTrigger trigger = pawn?.apparel.WornApparel
            .OfType<TrionTrigger>()
            .FirstOrDefault();

        if (trigger != null)
        {
            foreach (var mount in trigger.Mounts)
            {
                foreach (var comp in mount.EquippedComponents)
                {
                    componentDefNames.Add(comp.Def.defName);
                }
            }
        }
    }

    Scribe_Collections.Look(ref componentDefNames, "componentDefNames");

    if (Scribe.mode == LoadSaveMode.LoadingVars)
    {
        _cachedComponentDefNames = componentDefNames;
    }
}

private List<string> _cachedComponentDefNames = new List<string>();

public override void PostSpawnSetup(bool respawningAfterLoad)
{
    base.PostSpawnSetup(respawningAfterLoad);

    if (respawningAfterLoad && _cachedComponentDefNames?.Count > 0)
    {
        // 读档：重建组件对象
        Pawn pawn = parent as Pawn;
        TrionTrigger trigger = pawn?.apparel.WornApparel
            .OfType<TrionTrigger>()
            .FirstOrDefault();

        if (trigger != null)
        {
            // 简单策略：装到第一个有空位的槽位
            foreach (var defName in _cachedComponentDefNames)
            {
                var def = DefDatabase<ThingDef>.GetNamed(defName);
                var comp = TriggerComponent.CreateFromDef(def);

                foreach (var mount in trigger.Mounts)
                {
                    if (mount.HasSlotSpace && mount.TryAddComponent(comp))
                        break;
                }
            }
        }

        _cachedComponentDefNames.Clear();
    }
}
```

## 常见问题排查

### 问题1：wearer 为 null
**原因**：Apparel 还未被穿戴
**解决**：确保在 Pawn 穿戴后再访问 wearer，或通过 `pawn.apparel.WornApparel` 查询

### 问题2：GetWearerCompTrion() 返回 null
**原因**：Pawn 没有 CompTrion（可能未植入腺体）
**解决**：检查 Hediff_TrionGland 是否正确添加 CompTrion

### 问题3：TriggerComponent 的参数读不到
**原因**：ModExtension 序列化出问题
**解决**：检查 XML 中 ModExtension 的类名是否正确，确保 DefModExt 继承正确

### 问题4：破裂时快照不恢复
**原因**：_snapshot 为 null
**解决**：检查激活时快照是否正确捕获，确保序列化时保存

### 问题5：传送位置无效
**原因**：RCellFinder.RandomSpotJustOutsideColony 返回 Invalid
**解决**：检查地图上是否有建筑物（基地需要存在），或添加保护逻辑
