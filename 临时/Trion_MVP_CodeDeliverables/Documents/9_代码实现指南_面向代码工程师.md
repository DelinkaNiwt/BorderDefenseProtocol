### 1.1 TrionTrigger 类实现（Apparel子类）

**文件位置**：`ProjectTrion_MVP/TrionTrigger.cs`

**重要**：TrionTrigger 现在继承自 Apparel（不是 ThingWithComps），复用穿戴系统。

```csharp
using System.Collections.Generic;
using Verse;
using RimWorld;

namespace ProjectTrion_MVP
{
    /// <summary>
    /// Trion触发器物品 - 继承Apparel，可装配多个TriggerComponent组件
    /// 通过穿戴系统进行装备/脱下管理（pawn.apparel.Wear/Unfold）
    /// </summary>
    public class TrionTrigger : Apparel
    {
        // ============ 数据字段 ============
        private List<TriggerMount> _mounts = new List<TriggerMount>();

        // ============ 公开属性 ============
        public List<TriggerMount> Mounts => _mounts;
        public int MountCount => _mounts.Count;

        public override void SpawnSetup(Map map, bool respawningAfterLoad)
        {
            base.SpawnSetup(map, respawningAfterLoad);

            // 初始化时创建3个挂载点（左手、右手、特殊）
            if (!respawningAfterLoad && _mounts.Count == 0)
            {
                _mounts.Add(new TriggerMount(this, "LeftHand", 4));    // 左手槽位数=4
                _mounts.Add(new TriggerMount(this, "RightHand", 4));   // 右手槽位数=4
                _mounts.Add(new TriggerMount(this, "Special", 1));     // 特殊槽位数=1

                Log.Message("[Trion] 触发器生成，挂载点初始化完成");
            }
        }

        /// <summary>
        /// 获取穿着者（Pawn装备了此触发器）
        /// 穿戴系统会自动管理这个关系
        /// </summary>
        public Pawn GetWearer()
        {
            return wearer;  // Apparel系统提供的属性
        }

        /// <summary>
        /// 获取穿着者的CompTrion
        /// </summary>
        public CompTrion GetWearerCompTrion()
        {
            return wearer?.GetComp<CompTrion>();
        }

        /// <summary>
        /// 获取指定挂载点
        /// </summary>
        public TriggerMount GetMount(string slotName)
        {
            foreach (var mount in _mounts)
            {
                if (mount.SlotName == slotName)
                    return mount;
            }
            return null;
        }

        /// <summary>
        /// 序列化数据（Apparel会自动序列化穿戴状态）
        /// 我们只需要序列化自定义的_mounts列表
        /// </summary>
        public override void ExposeData()
        {
            base.ExposeData();
            Scribe_Collections.Look(ref _mounts, "mounts", LookMode.Deep);
        }
    }
}
```

**注意**：
- TrionTrigger 现在继承自 **Apparel**，不是 ThingWithComps
- 穿戴关系通过 `pawn.apparel.Wear(trigger)` 自动管理
- `wearer` 属性由 Apparel 提供（表示穿着者）
- 存档/读档由 Apparel 系统自动处理穿戴状态

```csharp
using System.Collections.Generic;
using Verse;

namespace ProjectTrion_MVP
{
    /// <summary>
    /// 触发器挂载点 - 管理组件装备、激活、消耗
    /// </summary>
    public class TriggerMount : IExposable
    {
        // ============ 数据字段 ============
        private TrionTrigger _parent;
        private string _slotName; // "LeftHand" / "RightHand" / "Special"
        private int _maxSlots; // 最大槽位数（左/右=4，特殊=1）
        private List<TriggerComponent> _equippedComponents = new List<TriggerComponent>();
        private TriggerComponent _currentlyActive; // 当前激活的组件（每个槽位最多1个激活）

        // ============ 构造函数 ============
        public TriggerMount()
        {
            // 用于序列化
        }

        public TriggerMount(TrionTrigger parent, string slotName, int maxSlots)
        {
            _parent = parent;
            _slotName = slotName;
            _maxSlots = maxSlots;
        }

        // ============ 公开属性 ============
        public string SlotName => _slotName;
        public List<TriggerComponent> EquippedComponents => _equippedComponents;
        public TriggerComponent CurrentlyActive => _currentlyActive;
        public int UsedSlots => _equippedComponents.Count;
        public bool HasSlotSpace => UsedSlots < _maxSlots;

        // ============ 核心方法 ============

        /// <summary>
        /// 尝试装备组件
        /// </summary>
        public bool TryAddComponent(TriggerComponent comp)
        {
            // 检查1: 槽位是否足够
            if (!HasSlotSpace)
            {
                Log.Warning($"[Trion] 槽位不足: {_slotName}");
                return false;
            }

            // 检查2: 是否冲突（同类型组件只能装1个）
            if (!CanCoexistWith(comp))
            {
                Log.Warning($"[Trion] 组件冲突: {comp.def.label}");
                return false;
            }

            // 检查3: 战斗体是否激活（激活后不能修改）
            CompTrion wearerComp = _parent.GetWearerCompTrion();  // 改用GetWearer获取CompTrion
            if (wearerComp?.IsInCombat ?? false)
            {
                Log.Warning("[Trion] 战斗体已激活，无法修改配置");
                return false;
            }

            // 装备组件
            _equippedComponents.Add(comp);

            // 如战斗体已激活则需要重新计算Reserved
            if (wearerComp != null && wearerComp.IsInCombat)
            {
                // 这里会在战斗体激活后调用，重新计算Reserved值
                // wearerComp.RecalculateReserved();
            }

            Log.Message($"[Trion] 组件装备成功: {comp.def.label} → {_slotName}");
            return true;
        }

        /// <summary>
        /// 检查组件是否可以共存
        /// </summary>
        private bool CanCoexistWith(TriggerComponent comp)
        {
            // 简单规则：不检查冲突（MVP阶段暂不实现组件冲突）
            // 复杂规则可在此添加
            return true;
        }

        /// <summary>
        /// 激活组件
        /// </summary>
        public void SetActiveComponent(TriggerComponent comp)
        {
            if (!_equippedComponents.Contains(comp))
            {
                Log.Warning("[Trion] 尝试激活未装备的组件");
                return;
            }

            // 旧组件关闭
            if (_currentlyActive != null)
            {
                _currentlyActive.SetActive(false);
            }

            // 新组件激活
            _currentlyActive = comp;
            comp.SetActive(true);
            Log.Message($"[Trion] 组件激活: {comp.def.label}");
        }

        /// <summary>
        /// 获取总消耗（所有激活组件）
        /// </summary>
        public float GetTotalConsumption()
        {
            float total = 0;
            foreach (var comp in _equippedComponents)
            {
                if (comp.IsActive)
                {
                    total += comp.TrionCost; // TrionCost = 每60Tick消耗
                }
            }
            return total;
        }

        /// <summary>
        /// 获取占用值（所有装备组件）
        /// </summary>
        public float GetTotalReserved()
        {
            float total = 0;
            foreach (var comp in _equippedComponents)
            {
                total += comp.ReservedCost; // ReservedCost = 占用值
            }
            return total;
        }

        // ============ 序列化 ============
        public void ExposeData()
        {
            Scribe_Values.Look(ref _slotName, "slotName");
            Scribe_Values.Look(ref _maxSlots, "maxSlots");
            Scribe_Collections.Look(ref _equippedComponents, "components", LookMode.Deep);
            Scribe_References.Look(ref _currentlyActive, "currentlyActive");
        }
    }
}
```

```csharp
using Verse;

namespace ProjectTrion_MVP
{
    /// <summary>
    /// 触发器组件基类 - 所有组件的父类
    /// </summary>
    public class TriggerComponent : ThingWithComps
    {
        // ============ 数据字段 ============
        protected bool _isActive = false;
        protected TriggerComponent_Def _componentDef;

        public TriggerComponent_Def ComponentDef => def as TriggerComponent_Def;

        // ============ 属性 ============
        public bool IsActive => _isActive;

        public float ReservedCost => ComponentDef?.reservedCost ?? 10f;
        public float TrionCost => ComponentDef?.trionCostPerTick ?? 1f;
        public float ActivationCost => ComponentDef?.activationCost ?? 5f;

        // ============ 核心方法 ============

        public virtual void SetActive(bool active)
        {
            _isActive = active;
            Log.Message($"[Trion] {def.label} 设置激活状态: {active}");
        }

        public virtual void OnActivated()
        {
            // 激活时调用（子类覆写以实现特殊行为）
            Log.Message($"[Trion] {def.label} 已激活");
        }

        public virtual void OnDeactivated()
        {
            // 停用时调用
            Log.Message($"[Trion] {def.label} 已停用");
        }

        public override void ExposeData()
        {
            base.ExposeData();
            Scribe_Values.Look(ref _isActive, "isActive");
        }
    }

    /// <summary>
    /// TriggerComponent 定义（XML中使用）
```

```csharp
using Verse;
using RimWorld;

namespace ProjectTrion_MVP
{
    /// <summary>
    /// 触发器配置台 - 用于装配组件到触发器
    /// </summary>
    public class Building_TriggerConfigBench : Building
    {
        public override void ExposeData()
        {
            base.ExposeData();
        }

        /// <summary>
        /// 尝试为触发器装备组件
        /// </summary>
        public bool TryEquipComponent(TrionTrigger trigger, TriggerComponent component, string slotName)
        {
            if (trigger == null || component == null)
                return false;

            TriggerMount mount = trigger.GetMount(slotName);
            if (mount == null)
                return false;

            // 尝试装备
            bool success = mount.TryAddComponent(component);

            if (success)
            {
                // 从地图移除组件物品
                component.Destroy(DestroyMode.Vanish);
                Log.Message($"[Trion] 组件已装备到{slotName}");
            }

            return success;
        }
    }
}
```

## 实现步骤 4: CombatBodySnapshot 快照数据

**文件位置**：`ProjectTrion_MVP/CombatBodySnapshot.cs`

```csharp
using System.Collections.Generic;
using Verse;

namespace ProjectTrion_MVP
{
    /// <summary>
    /// 战斗体快照 - 保存生成战斗体前的Pawn状态
    /// </summary>
    public class CombatBodySnapshot : IExposable
    {
        // ============ 生命值数据 ============
        public float bodyHealthSum; // 身体总血量
        public Dictionary<BodyPartRecord, float> partHealth = new Dictionary<BodyPartRecord, float>();

        // ============ 部位状态 ============
        public List<BodyPartRecord> missingParts = new List<BodyPartRecord>();
        public List<Hediff_Injury> injuries = new List<Hediff_Injury>();

        // ============ 装备数据 ============
        public List<Apparel> wornApparel = new List<Apparel>();
        public List<ThingWithComps> inventory = new List<ThingWithComps>();

        // ============ 捕获快照 ============
        public void CaptureFromPawn(Pawn pawn)
        {
            if (pawn == null) return;

            // 1. 保存生命值
            bodyHealthSum = pawn.health.summaryHealth.SummaryHealthPercent;

            // 2. 保存各部位血量
            partHealth.Clear();
            foreach (var part in pawn.health.hediffSet.GetNotMissingParts())
            {
                float healthPercentage = pawn.health.hediffSet.GetPartHealth(part) / part.def.hitPoints;
                partHealth[part] = healthPercentage;
            }

            // 3. 保存缺失部位
            missingParts.Clear();
            missingParts.AddRange(pawn.health.hediffSet.GetMissingPartsCommonAncestors());

            // 4. 保存伤口
            injuries.Clear();
            foreach (var hediff in pawn.health.hediffSet.hediffs)
            {
                if (hediff is Hediff_Injury injury)
                {
                    injuries.Add(injury);
                }
            }

            // 5. 保存穿着装备
            wornApparel.Clear();
            wornApparel.AddRange(pawn.apparel.WornApparel);

            // 6. 保存背包物品
            inventory.Clear();
            if (pawn.inventory != null)
            {
                foreach (var thing in pawn.inventory.innerContainer)
                {
                    if (thing is ThingWithComps twc)
                    {
                        inventory.Add(twc);
                    }
                }
            }

            Log.Message($"[Trion] {pawn.Name} 快照已保存");
        }

        // ============ 恢复快照 ============
        public void RestoreToPawn(Pawn pawn)
        {
            if (pawn == null) return;

            // 1. 恢复血量
            // 注意：直接修改生命值比较复杂，需要通过HediffSet的API
            // 这里示意需要通过减少伤口来恢复

            // 2. 移除战斗体产生的新伤口
            // 保留快照中的伤口，移除不在快照中的伤口
            var currentInjuries = new List<Hediff>();
            foreach (var hediff in pawn.health.hediffSet.hediffs)
            {
                if (hediff is Hediff_Injury)
                {
                    currentInjuries.Add(hediff);
                }
            }

            foreach (var hediff in currentInjuries)
            {
                if (!injuries.Contains(hediff as Hediff_Injury))
                {
                    pawn.health.RemoveHediff(hediff);
                }
            }

            // 3. 恢复缺失部位
            // 部位恢复比较复杂，MVP阶段可能不完全实现

            // 4. 恢复装备
            pawn.apparel.WornApparel.Clear();
            foreach (var apparel in wornApparel)
            {
                pawn.apparel.Wear(apparel);
            }

            // 5. 恢复背包
            if (pawn.inventory != null && inventory.Count > 0)
            {
                foreach (var thing in inventory)
                {
                    pawn.inventory.innerContainer.TryAdd(thing);
                }
            }

            Log.Message($"[Trion] {pawn.Name} 快照已恢复");
        }

        // ============ 序列化 ============
        public void ExposeData()
        {
            Scribe_Values.Look(ref bodyHealthSum, "bodyHealthSum");
            Scribe_Collections.Look(ref partHealth, "partHealth", LookMode.Reference, LookMode.Value);
            Scribe_Collections.Look(ref missingParts, "missingParts", LookMode.Reference);
            Scribe_Collections.Look(ref injuries, "injuries", LookMode.Deep);
            Scribe_Collections.Look(ref wornApparel, "wornApparel", LookMode.Deep);
            Scribe_Collections.Look(ref inventory, "inventory", LookMode.Deep);
        }
    }
}
```

```csharp
using HarmonyLib;
using Verse;

namespace ProjectTrion_MVP.Patches
{
    /// <summary>
    /// Harmony补丁 - 拦截Pawn伤害，转化为Trion消耗
    /// </summary>
    [HarmonyPatch(typeof(Pawn), nameof(Pawn.TakeDamage))]
    public static class Patch_PawnTakeDamage
    {
        public static void Prefix(Pawn __instance, DamageInfo dinfo)
        {
            // 检查Pawn是否有CompTrion且已激活战斗体
            CompTrion comp = __instance.GetComp<CompTrion>();

            if (comp == null || !comp.IsInCombat)
                return;

            // 伤害转化为Trion消耗（1:1转化）
            float damageAmount = dinfo.Amount;

            // 消耗Trion
            comp.Consume(damageAmount);

            // 阻止肉身受伤
            dinfo.SetAmount(0);

            // 记录日志
            Log.Message($"[Trion] {__instance.Name} 受到伤害 {damageAmount}，转化为Trion消耗");
        }
    }
}
```

```csharp
/// </summary>
public void RecalculateReserved()
{

    Log.Message($"[Trion] Reserved已重新计算: {_reserved}");
}

/// <summary>
/// 每60Tick的消耗计算（关键方法）
/// </summary>
private void TickConsumption()
{
    if (!_isInCombat)
        return;

    if (!IsHashIntervalTick(60)) // 每60Tick执行一次
        return;

    // 步骤1: 基础维持消耗
    float baseMaintenance = _strategy.GetBaseMaintenance(); // 5f

    // 步骤2: 组件消耗
    float mountConsumption = GetMountsConsumption();

    // 步骤3: 泄漏速率（来自伤口）
    float leakRate = GetLeakRate();

    // 步骤4: 总消耗
    float totalConsumption = baseMaintenance + mountConsumption + leakRate;

    // 步骤5: 执行消耗
    Consume(totalConsumption);

    // 步骤6: 检查可用值是否枯竭
    if (Available <= 0)
    {
        TriggerBailOut();
    }

    // 步骤7: 策略回调
    _strategy.OnTick(this);
}
```

## 常见问题排查

### 问题1: "找不到类 XYZ"
**原因**：命名空间错误或文件位置错误
**解决**：检查 using 语句，确保类在 `ProjectTrion_MVP` 命名空间

### 问题2: 组件装备后没有出现在列表中
**原因**：TryAddComponent 返回 false
**排查**：
1. 检查 HasSlotSpace（槽位是否满）
2. 检查 CanCoexistWith（是否有冲突）
3. 检查 IsInCombat（是否战斗体已激活）

### 问题3: 序列化报错
**原因**：可能是嵌套数据结构的序列化问题
**解决**：
1. 确保所有字段都在 ExposeData 中处理
2. 使用正确的 Look 方法（Values/Deep/References）
3. 检查初始化顺序

### 问题4: Harmony补丁不生效
**原因**：补丁未正确注册或方法签名不匹配
**检查**：
1. 确保调用了 `harmony.PatchAll()`
2. 检查 `[HarmonyPatch]` 特性的参数
3. 验证 Prefix/Postfix 方法签名正确
