```csharp
// 在 ProjectTrion_MVP/Source/ProjectTrion_MVP/Components/TriggerMount.cs 中

public class TriggerMount
{
    // 新增字段
    private int guidanceTicksRemaining = 0;  // 导引剩余时间

    public void Activate(CompTrion comp)
    {
        if (state == MountState.Active)
            return;  // 已激活

        // 步骤1：检查激活费用
        if (def.activationCost > 0 && comp.Available < def.activationCost)
        {
            Messages.Message(
                $"Trion不足，无法激活{def.label}（需要{def.activationCost}，当前{comp.Available}）",
                MessageTypeDefOf.RejectInput
            );
            return;
        }

        // 步骤2：支付激活费用
        if (def.activationCost > 0)
        {
            comp.Consume(def.activationCost);
        }

        // 步骤3：进入激活导引状态
        if (def.activationGuidanceTicks > 0)
        {
            state = MountState.Activating;
            guidanceTicksRemaining = def.activationGuidanceTicks;
        }
        else
        {
            // 无导引时间，直接激活
            state = MountState.Active;
            Messages.Message($"{def.label}已激活", MessageTypeDefOf.PositiveEvent);
        }
    }

    public void Deactivate()
    {
        state = MountState.Dormant;
        guidanceTicksRemaining = 0;
    }

    public void Tick()
    {
        if (state == MountState.Activating)
        {
            guidanceTicksRemaining--;
            if (guidanceTicksRemaining <= 0)
            {
                state = MountState.Active;
                Messages.Message($"{def.label}激活完成", MessageTypeDefOf.PositiveEvent);
            }
        }
    }
}
```

```csharp
namespace ProjectTrion_MVP.Components
{
    public enum MountState
    {
        /// <summary>未连接：战斗体未生成，或挂载点被毁</summary>
```

```csharp
public class TriggerMount
{
    public MountState state = MountState.Disconnected;

    /// <summary>该组件是否占用Trion容量（Dormant、Activating、Active都占用）</summary>
    public bool IsReserving => state == MountState.Dormant ||
                               state == MountState.Activating ||
                               state == MountState.Active;

    /// <summary>该组件是否提供功能（仅Active提供）</summary>
    public bool IsActive => state == MountState.Active;

    // 更新CalculateTotalReserved()
    public float CalculateTotalReserved()
    {
        float total = 0f;
        foreach (var component in equippedComponents)
        {
            // 所有占用状态都计入Reserved
            if (component.Mount?.IsReserving ?? false)
                total += component.Mount.def.reservedCost;
        }
        return total;
    }

    // 更新CalculateTotalTrionCost()
    public float CalculateTotalTrionCost()
    {
        float total = 0f;
        foreach (var component in equippedComponents)
        {
            // 仅Active状态才产生消耗
            if (component.Mount?.IsActive ?? false)
                total += component.Mount.def.consumptionRate;
        }
        return total;
    }
}
```

```csharp
public class TriggerMount
{
    public SlotType slotType;

    /// <summary>检查是否可以激活新组件（考虑同时激活数量限制）</summary>
    public bool CanActivateNew(TriggerComponent component)
    {
        // 特殊槽位无限制
        if (slotType == SlotType.Special)
            return true;

        // 手部槽位：检查是否有已激活的组件
        var activeCount = equippedComponents.Count(c => c.Mount?.IsActive ?? false);
        return activeCount == 0;  // 仅当没有激活组件时允许激活新的
    }

    /// <summary>获取该槽位中已激活的组件</summary>
    public List<TriggerComponent> GetActiveComponents()
    {
        return equippedComponents.Where(c => c.Mount?.IsActive ?? false).ToList();
    }
}
```

**步骤3** - TrionTrigger实现激活切换
```csharp
public class TrionTrigger : Apparel
{
    private CompTrion _compTrion;

    /// <summary>激活指定组件，如有必要则关闭同槽位其他组件</summary>
    public void TryActivateComponent(TriggerComponent component)
    {
        var mount = component.Mount;
        if (mount == null)
            return;

        // 检查是否可以直接激活
        if (mount.CanActivateNew(component))
        {
            mount.Activate(_compTrion);
            return;
        }

        // 否则启动激活切换流程
        StartActivationSwitch(mount, component);
    }

    /// <summary>激活切换：旧组件→新组件</summary>
    private void StartActivationSwitch(TriggerMount mount, TriggerComponent newComponent)
    {
        var activeComponents = mount.GetActiveComponents();

        if (activeComponents.Count > 0)
        {
            var oldComponent = activeComponents[0];

            Messages.Message(
                $"切换{mount.slotType}的组件：{oldComponent.Mount.def.label} → {newComponent.Mount.def.label}",
                MessageTypeDefOf.NeutralEvent
            );

            // 新组件进入激活导引
            newComponent.Mount.Activate(_compTrion);

            // 旧组件保持激活状态，直到导引完成
            // （由Tick()在导引完成后关闭旧组件）
        }
    }

    public void Tick_ActivationSwitch()
    {
        // 每个槽位检查是否有"激活中"组件
        foreach (var mount in GetAllMounts())
        {
            if (mount.slotType == SlotType.Special)
                continue;  // 特殊槽位无需检查

            var activatingList = mount.equippedComponents
                .Where(c => c.Mount?.state == MountState.Activating)
                .ToList();

            // 如果有激活中的组件，检查导引是否完成
            foreach (var newComponent in activatingList)
            {
                if (newComponent.Mount.guidanceTicksRemaining <= 0)
                {
                    // 导引完成，关闭其他激活的组件
                    var oldActive = mount.GetActiveComponents();
                    foreach (var old in oldActive)
                    {
                        if (old != newComponent)
                        {
                            old.Mount.Deactivate();
                        }
                    }
                }
            }
        }
    }
}
```

**验证方式**：

- [ ] 特殊槽位可同时激活多个
- [ ] 激活新组件时自动关闭旧的

```csharp
            VitalPartUtil.IsLeg(missingPart.Part))
```

### ✅ 修改7：Bail Out前置检查

```csharp
// DefaultTrionStrategy.cs 中修改 CanBailOut 方法

public override bool CanBailOut(CompTrion comp)
{
    var pawn = comp.pawn;
    var trigger = pawn.apparel.FirstOrDefault(a => a is TrionTrigger) as TrionTrigger;

    if (trigger == null)
        return false;  // 未装备触发器

    // 检查是否装备了"脱离系统"组件
    var bailOutMount = trigger.GetAllMounts()
        .SelectMany(m => m.equippedComponents)
        .FirstOrDefault(c => c.def.defName == "TrionComponent_BailOut");

    if (bailOutMount == null)
    {
        // 未装备脱离系统组件
        return false;
    }

    // 检查脱离系统组件是否激活（或至少处于Dormant状态）
    if (bailOutMount.Mount.state == MountState.Disconnected)
    {
        return false;
    }

    // 检查传送目标有效性
    return GetBailOutTarget(comp) != IntVec3.Invalid;
}

public override IntVec3 GetBailOutTarget(CompTrion comp)
{
    var pawn = comp.pawn;
    var map = pawn.Map;

    // 优先级1：查询Building_BailOutAnchor
    var anchors = map.listerThings.ThingsOfDef(
        DefDatabase<ThingDef>.GetNamed("Building_BailOutAnchor", false)
    );

    if (anchors != null && anchors.Count > 0)
    {
        // 找最近的激活锚点
        var activeAnchors = anchors.OfType<Building>()
            .Where(b => !b.FloorTouched || b.Destroyed == false)
            .ToList();

        if (activeAnchors.Count > 0)
        {
            var nearest = activeAnchors
                .OrderBy(a => pawn.Position.DistanceTo(a.Position))
                .First();
            return nearest.Position;
        }
    }

    // 优先级2：地图边缘（应急传送）
    var edgeCell = CellFinder.RandomEdgeCell(map);
    if (edgeCell.Walkable(map))
        return edgeCell;

    // 失败
    return IntVec3.Invalid;
}
```

- [ ] 装备脱离系统后，CanBailOut返回true
- [ ] Bail Out正确传送到锚点或边缘

```csharp
namespace ProjectTrion_MVP.Components
{
    /// <summary>管理能力使用和功率检查</summary>
    public static class AbilityController
    {
        /// <summary>检查是否可以使用指定能力</summary>
        public static bool CanUseAbility(Ability ability, CompTrion comp)
        {
            if (ability.RequiredMinPower <= 0)
                return true;  // 无功率要求

            float power = comp.OutputPower;
            if (power < ability.RequiredMinPower)
            {
                Messages.Message(
                    $"{ability.label}需要输出功率≥{ability.RequiredMinPower}，当前{power}",
                    MessageTypeDefOf.RejectInput
                );
                return false;
            }

            // 检查Trion消耗
            if (comp.Available < ability.TrionCost)
            {
                Messages.Message(
                    $"Trion不足以释放{ability.label}",
                    MessageTypeDefOf.RejectInput
                );
                return false;
            }

            return true;
        }

        /// <summary>使用能力（消耗Trion）</summary>
        public static void UseAbility(Ability ability, CompTrion comp)
        {
            comp.Consume(ability.TrionCost);
            Messages.Message(
                $"{comp.pawn.LabelCap}释放了{ability.label}",
                MessageTypeDefOf.PositiveEvent
            );
        }
    }

    /// <summary>能力定义</summary>
    public class Ability
    {
        public string label;
        public float RequiredMinPower;  // 最低功率要求
        public float TrionCost;         // Trion消耗
        public float Cooldown;          // 冷却时间

        // 效果相关
        public float Range;             // 作用范围（会根据功率调整）
        public float Damage;            // 伤害（会根据功率调整）
        public float Duration;          // 持续时间Tick数（会根据功率调整）
```

```csharp
namespace ProjectTrion_MVP.Components
{
    /// <summary>根据输出功率调整组件表现</summary>
    public static class ComponentPowerScaling
    {
        /// <summary>根据功率获取武器伤害倍数</summary>
        public static float GetWeaponDamageMultiplier(CompTrion comp)
        {
            float power = comp.OutputPower;
            return power switch
            {
                < 20 => 0.7f,   // E级：-30%伤害
                < 30 => 0.85f,  // D级：-15%伤害
                < 40 => 1.0f,   // C级：基准
                < 50 => 1.15f,  // B级：+15%伤害
                < 70 => 1.3f,   // A级：+30%伤害
                >= 70 => 1.5f,  // S级：+50%伤害
            };
        }

        /// <summary>根据功率获取AOE范围倍数</summary>
        public static float GetAOERangeMultiplier(CompTrion comp)
        {
            float power = comp.OutputPower;
            return 0.5f + (power / 100f) * 1.5f;  // 0.5-2.0倍范围
        }

        /// <summary>根据功率获取护盾减伤百分比</summary>
        public static float GetShieldMitigationPercent(CompTrion comp)
        {
            float power = comp.OutputPower;
            return Mathf.Clamp01(power / 100f * 0.8f);  // 0-80%减伤
        }

        /// <summary>根据功率获取特殊效果持续时间倍数</summary>
        public static float GetEffectDurationMultiplier(CompTrion comp)
```
