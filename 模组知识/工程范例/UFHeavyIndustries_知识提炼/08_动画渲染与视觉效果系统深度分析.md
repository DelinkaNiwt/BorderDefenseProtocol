## 2. SRA框架基础

### 2.1 什么是SRA？

**SRA** = Shader Rendering Animation

高性能渲染库，针对RimWorld的视觉效果优化。主要特性：
- **GPU驱动**：计算在GPU上进行
- **批处理**：减少Draw Call
- **动画优化**：高效的关键帧插值
- **着色器库**：预制常见效果

### 2.2 SRA集成

```csharp
using SRA;

public class CompAnimationRenderer : ThingComp
{
    // SRA渲染器实例
    private SRA_AnimationRenderer sraRenderer;

    // 初始化
    public override void Initialize(CompProperties props)
    {
        base.Initialize(props);

        // 创建SRA渲染器
        sraRenderer = new SRA_AnimationRenderer(parent);

        // 注册着色器
        sraRenderer.RegisterShader("Shield", ShaderDefOf.Transparent);
        sraRenderer.RegisterShader("LaserBeam", ShaderDefOf.Cutout);
    }

    // 更新动画
    public override void CompTick()
    {
        base.CompTick();

        // 推进动画时间
        sraRenderer.UpdateAnimationTime(1f / 60f);

        // 更新顶点属性
        sraRenderer.UpdateVertexBuffer();
    }

    // 渲染（由RimWorld自动调用）
    public void RenderMesh()
    {
        sraRenderer.Render();
    }
}
```

## 3. Mote粒子系统

### 3.1 Mote基础

**Mote** = 轻量级动画粒子，用于特效。

```csharp
public abstract class Mote : Thing
{
    // 基本属性
    protected float scale = 1f;
    protected Color color = Color.white;
    protected Vector3 velocity = Vector3.zero;

    // 生命周期
    protected int ticksToLive;  // 还能活多少tick
    protected int totalTicks;   // 总寿命

    // 渲染
    protected Graphic graphic;
    protected float rotationRate = 0f;

    public override void Tick()
    {
        base.Tick();

        // 更新位置
        exactPosition += velocity * 0.0166f;  // tick转换为秒

        // 衰减
        ticksToLive--;
        if (ticksToLive <= 0)
        {
            Destroy();
        }

        // 更新视觉
        UpdateAlpha();
        UpdateScale();
    }

    protected virtual void UpdateAlpha()
    {
        // 淡出效果：后期变透明
        float progress = 1f - (float)ticksToLive / totalTicks;
        color.a = Mathf.Lerp(1f, 0f, progress);
    }

    protected virtual void UpdateScale()
    {
        // 可选：粒子变小
        scale = Mathf.Lerp(1f, 0f, progress);
    }
}
```

```csharp
public class MotePool : MonoBehaviour

    private List<Mote> smokePool = new();
    private List<Mote> dustPool = new();

            smokePool.Add(smoke);
        }
    }

    {
        if (smokePool.Count > 0)
        {
            // 复用
            var mote = smokePool[smokePool.Count - 1];
            smokePool.RemoveAt(smokePool.Count - 1);
            return mote;
        }

        // 池空了，新建
        return (Mote)ThingMaker.MakeThing(ThingDefOf.Mote_Smoke);
    }

    public void ReturnSmoke(Mote mote)
    {
        // 重置状态
        mote.velocity = Vector3.zero;
        mote.scale = 1f;
        mote.color = Color.white;

        // 返回池
        smokePool.Add(mote);
    }
}
```

## 4. 动画系统架构

### 4.1 关键帧动画

```csharp
public class AnimationDef
{
    public string defName;
    public Graphic baseGraphic;

    // 关键帧列表
    public List<KeyFrame> keyFrames = new();

    public struct KeyFrame
    {
        public float timePercent;  // 动画进度百分比
        public Graphic graphic;    // 该帧的图形
        public Vector3 scale;      // 缩放
        public float rotation;     // 旋转
        public Color color;        // 颜色/透明度
    }
}

public class AnimationController
{
    private AnimationDef currentAnimation;
    private float animationProgress = 0f;
    private float animationSpeed = 1f;
    private bool isLooping = false;

    public void PlayAnimation(AnimationDef anim, bool loop = false)
    {
        currentAnimation = anim;
        animationProgress = 0f;
        isLooping = loop;
    }

    public void Update(float deltaTime)
    {
        if (currentAnimation == null)
            return;

        // 推进进度
        animationProgress += deltaTime * animationSpeed;

        // 检查是否完成
        if (animationProgress >= 1f)
        {
            if (isLooping)
            {
                animationProgress -= 1f;  // 循环
            }
            else
            {
                currentAnimation = null;  // 停止
            }
        }
    }

    public KeyFrame GetCurrentFrame()
    {
        // 二分查找相邻的两个关键帧
        int frameIndex = FindFrameIndex(animationProgress);

        KeyFrame prev = currentAnimation.keyFrames[frameIndex];
        KeyFrame next = currentAnimation.keyFrames[frameIndex + 1];

        // 插值
        float localProgress = (animationProgress - prev.timePercent)
                            / (next.timePercent - prev.timePercent);

        return Lerp(prev, next, localProgress);
    }
}
```

### 5.2 护盾激活/关闭动画

```csharp
public class ShieldActivationEffect
{
    public IEnumerator PlayActivation(Vector3 centerPos, float finalRadius, Map map)
    {
        float duration = 0.5f;
        float elapsed = 0f;

        while (elapsed < duration)
        {
            elapsed += Time.deltaTime;

            // 扩张动画
            float progress = elapsed / duration;
            float currentRadius = finalRadius * Mathf.Pow(progress, 2);  // 加速扩张

            // 逐帧生成扩张粒子
            SpawnExpansionParticles(centerPos, currentRadius);

            // 播放声音（仅一次）
            if (progress == 0)
                SoundDefOf.ShieldActivate.PlayOneShot(centerPos);

            yield return null;
        }
    }

    public IEnumerator PlayDeactivation(Vector3 centerPos, float initialRadius, Map map)
    {
        float duration = 0.3f;
        float elapsed = 0f;

        while (elapsed < duration)
        {
            elapsed += Time.deltaTime;

            // 收缩动画
            float progress = elapsed / duration;
            float currentRadius = initialRadius * (1f - Mathf.Pow(progress, 2));

            // 逐帧生成收缩粒子
            SpawnContractionParticles(centerPos, currentRadius);

            yield return null;
        }
    }

    private void SpawnExpansionParticles(Vector3 center, float radius)
    {
        // 在球面上生成粒子
        for (int i = 0; i < 8; i++)
        {
            float angle = (float)i / 8 * Mathf.PI * 2;
            Vector3 pos = center + new Vector3(Mathf.Cos(angle), 0, Mathf.Sin(angle)) * radius;

            MoteMaker.ThrowExplosionFlash(pos, parent.Map, 0.5f);
        }
    }
}
```

```csharp
public class TurretAnimator
{
    private float targetRotation = 0f;
    private float currentRotation = 0f;
    private float rotationSpeed = 45f;  // 度/秒

    public void UpdateRotation(float deltaTime, LocalTargetInfo target)
    {
        if (!target.IsValid)
            return;

        // 计算目标旋转角
        Vector3 direction = (target.CenterVector3 - parent.Position.ToVector3()).normalized;

        currentRotation = Mathf.MoveTowardsAngle(
            currentRotation,
            targetRotation,
            rotationSpeed * deltaTime
        );

        // 应用旋转到炮身
        barrelObject.transform.localRotation = Quaternion.AngleAxis(currentRotation, Vector3.up);
    }
}
```

```csharp
public class Beam : Thing
{
    // 起点和终点
    public Vector3 origin;
    public Vector3 target;

    // 外观
    public float width = 0.5f;
    public Color color = Color.cyan;

    // 动画
    public float animationProgress = 0f;
    public float animationSpeed = 1f;

    public override void Tick()
    {
        base.Tick();

        // 更新动画进度
        animationProgress += animationSpeed * 0.0166f;
        if (animationProgress >= 1f)
        {
            // 激光完成，销毁
            Destroy();
        }
    }

    public override void Draw()
    {
        // 绘制激光束
        DrawBeam(origin, target, width, color);
    }

    private void DrawBeam(Vector3 start, Vector3 end, float beamWidth, Color beamColor)
    {
        // 使用LineRenderer或custom mesh
        // 这里使用简化版本

        Vector3 direction = (end - start).normalized;
        Vector3 perpendicular = Vector3.Cross(direction, Vector3.up).normalized;

        // 创建四边形来表示激光
        Vector3 topLeft = start + perpendicular * beamWidth * 0.5f;
        Vector3 topRight = end + perpendicular * beamWidth * 0.5f;
        Vector3 bottomLeft = start - perpendicular * beamWidth * 0.5f;
        Vector3 bottomRight = end - perpendicular * beamWidth * 0.5f;

        // 绘制
        Gizmos.color = beamColor;
        Gizmos.DrawLine(topLeft, topRight);
        Gizmos.DrawLine(bottomLeft, bottomRight);
    }
}
```

```csharp
public class ExplosionEffectStack
{
    public static void PlayExplosion(Vector3 position, Map map, float radius)
    {
        // ① 中心闪光
        MoteMaker.ThrowExplosionFlash(position, map, radius * 0.3f);

        for (float r = 0.5f; r <= radius; r += 2f)
```

```csharp
public class ShockwaveEffect
{
    public IEnumerator PlayShockwave(Vector3 center, float maxRadius, Map map)
    {
        float duration = 0.5f;
        float elapsed = 0f;

        while (elapsed < duration)
        {
            elapsed += Time.deltaTime;
            float progress = elapsed / duration;

            // 波前半径
            float waveRadius = maxRadius * progress;

            // 波动强度（高斯分布）
            float centerDist = Mathf.Abs(progress - 0.5f);
            float intensity = Mathf.Exp(-centerDist * centerDist * 10f);

            // 绘制波前环
            RenderWaveFront(center, waveRadius, intensity, map);

            yield return null;
        }
    }

    private void RenderWaveFront(Vector3 center, float radius, float intensity, Map map)
    {
        // 使用环形粒子表示冲击波
        int segments = Mathf.RoundToInt(radius * 8);
        float angleStep = Mathf.PI * 2 / segments;

        for (int i = 0; i < segments; i++)
        {
            float angle = angleStep * i;
            Vector3 pos = center + new Vector3(Mathf.Cos(angle), 0, Mathf.Sin(angle)) * radius;

            // 创建粒子
            var mote = (Mote)ThingMaker.MakeThing(ThingDefOf.Mote_Flash);
            mote.SetPosition(pos);
            mote.scale = 0.3f * intensity;

            GenSpawn.Spawn(mote, pos.ToIntVec3(), map);
        }
    }
}
```

```csharp
public void RenderMotes_Batched()
{
    // 按材质分组
    Dictionary<Material, List<Matrix4x4>> batches = new();

    foreach (var mote in allMotes)
    {
        if (!batches.TryGetValue(mote.material, out var matrices))
        {
            matrices = new List<Matrix4x4>();
            batches[mote.material] = matrices;
        }
        matrices.Add(mote.matrix);
    }

    // 批量渲染
    foreach (var batch in batches)
    {
        Graphics.DrawMeshInstanced(
            mesh,
            0,
            batch.Key,
            batch.Value,
            batch.Value.Count
        );
    }
}
```

```
1. 检查Draw Call数量
   if (Graphics.drawCalls > 500)

- 减少粒子生成速率
- 启用批处理
- 使用简化的着色器
```
